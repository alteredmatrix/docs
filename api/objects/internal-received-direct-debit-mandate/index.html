InternalReceivedDirectDebitMandate

Received direct debit mandate for SEPA

  version: String!
  createdAt: DateTime!
  updatedAt: DateTime!
  reference: String!
  scheme: InternalReceivedDirectDebitMandateScheme!

  executedAt: DateTime
  signatureDate: Date
  expiredAt: Date
  statusInfo: ReceivedDirectDebitMandateStatusInfo!
  account: Account
  iban: IBAN!
  name: String

}
